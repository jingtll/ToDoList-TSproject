### 7. DAO 层 - 数据访问

```typescript
// dao/noteDao.ts
export class noteDao implements Types.noteDao {
    noteModel = NoteModel

    // 分页查询
    async getNoteList(page: number, size: number) {
        try {
            const data: Array<Types.Note> = await this.noteModel.find({})
                .skip((page - 1) * size)
                .limit(size);
            return data;
        } catch (error) {
            return 0;
        }
    }

    // 条件查询 - 使用正则匹配
    async getNoteListByContent(content: string) {
        try {
            const data: Array<Types.Note> = await this.noteModel.find({
                content: new RegExp(`${content}`, 'g')
            });
            return data;
        } catch (error) {
            return 0;
        }
    }

    // 新增
    async addNote(note: Types.Note) {
        try {
            const noteIns = new this.noteModel(note);
            const data = await noteIns.save();
            return data;
        } catch (error) {
            return 0;
        }
    }

    // 删除
    async deleteNote(id: string) {
        try {
            const data = await this.noteModel.deleteOne({ _id: id });
            return data;
        } catch (error) {
            return 0;
        }
    }

    // 更新
    async updateNote(id: string, note: Types.Note) {
        try {
            const data = await this.noteModel.updateOne({ _id: id }, note);
            return data;
        } catch (error) {
            return 0;
        }
    }
}
```

## 第三部分：瀑布流布局实现详解

### 1. 核心思想

瀑布流布局的特点：
- 多列等宽不等高
- 元素依次填充到高度最小的列
- 视觉效果像瀑布一样错落有致

### 2. 实现策略

本项目采用 **"隐藏测量 + 双列分配"** 的策略：

```
┌─────────────────────────────────────────┐
│  init-list (隐藏的测量容器)              │
│  ┌─────┐ ┌─────┐ ┌─────┐ ┌─────┐       │
│  │  1  │ │  2  │ │  3  │ │  4  │ ...   │
│  └─────┘ └─────┘ └─────┘ └─────┘       │
│  测量每个元素高度，分配到左右两列         │
└─────────────────────────────────────────┘
                    ↓
┌───────────────────┬───────────────────┐
│    list-left      │    list-right     │
│  ┌─────┐          │  ┌─────┐          │
│  │  1  │          │  │  2  │          │
│  │h=80 │          │  │h=120│          │
│  └─────┘          │  └─────┘          │
│  ┌─────┐          │  ┌─────┐          │
│  │  3  │          │  │  4  │          │
│  │h=100│          │  │h=90 │          │
│  └─────┘          │  └─────┘          │
└───────────────────┴───────────────────┘
```

### 3. 完整代码解析

```vue
<script setup lang="ts">
import { getNotes } from '@/api/node'
import { nextTick, onMounted, reactive, ref, watch } from 'vue'
import type { Note, NoteList, NoteListState } from '@/types'

// 存储所有笔记数据
const notes = ref([] as NoteList)
// 存储 DOM 元素引用
const items = ref([] as HTMLElement[])

// 1️⃣ 从后端获取数据
const initList = () => {
  getNotes<NoteList>(1, 10).then((res) => {
    notes.value = res  // 触发 watch
  })
}

// 2️⃣ 定义左右两列的状态
const noteListState: NoteListState = {
  leftList: [],
  rightList: [],
}
const state = reactive(noteListState)

// 3️⃣ 核心算法：测量高度并分配到左右列
const initLRlist = () => {
  let leftHeightSum = 0    // 左列累计高度
  let rightHeightSum = 0   // 右列累计高度
  let leftArr: Note[] = []  // 左列数据
  let rightArr: Note[] = [] // 右列数据

  nextTick(() => {
    // 遍历隐藏容器中的所有元素
    items.value!.forEach((item, index) => {
      if (leftHeightSum <= rightHeightSum) {
        // 左列较矮，放左边
        leftArr.push(notes.value[index])
        leftHeightSum += item?.clientHeight
      } else {
        // 右列较矮，放右边
        rightArr.push(notes.value[index])
        rightHeightSum += item?.clientHeight
      }
    })
    // 反转数组（最新数据显示在上面）
    state.leftList = leftArr.reverse()
    state.rightList = rightArr.reverse()
  })
}

// 4️⃣ 生命周期：组件挂载后获取数据
onMounted(() => {
  initList()
})

// 5️⃣ 监听数据变化，重新分配
watch(notes, () => {
  initLRlist()
})
</script>
```

## 第六部分：钩子函数与防抖函数详解

### 6.1 整体概述 (Overview)

**文件用途：**
- `src/hooks/useLoadMore.ts` — 封装滚动加载逻辑的自定义组合式函数（Composable），用于实现触底加载更多数据
- `src/utils/debounce.ts` — 通用防抖工具函数，用于限制高频调用（如搜索输入）

**技术栈：**
- Vue 3 Composition API（`Ref`、`onUnmounted`）
- TypeScript 类型约束
- 原生 DOM API（`addEventListener`、`scroll` 事件）
- JavaScript 闭包与高阶函数

**运行逻辑：**
```
用户输入搜索 → watch 监听 → debounce 包装 → 等待 1000ms → handleSearch() → API 请求
用户滚动列表 → scroll 事件 → useLoadMore 判断触底 → loadMore() → 加载下一页
```

### 6.2 结构拆解 (Structure Breakdown)

**模块导入：**

| 文件 | 导入 | 来源 | 作用 |
|------|------|------|------|

### 6.3 逐段解释 (Line-by-Line Explanation)

#### useLoadMore.ts

```typescript
import { Ref } from 'vue'
// 从 Vue 导入 Ref 类型，用于约束参数必须是响应式引用

export default function useLoadMore(
  element: Ref<null | HTMLElement>,  // 滚动容器的 DOM 引用
  fn: Function                       // 触底时执行的回调
) {
  function loadMore() {
    const containerHeight = element.value?.clientHeight || 0
    // 可视区域高度（用户能看到的部分）

    const scrollTop = element.value?.scrollTop || 0
    // 已滚动的距离（被卷上去的高度）

    const scrollHeight = element.value?.scrollHeight || 0
    // 内容总高度（包括不可见部分）

    if (containerHeight + scrollTop + 10 >= scrollHeight) {
      fn()  // 距离底部不足 10px 时触发回调
    }
  }

  element.value?.addEventListener('scroll', loadMore)
  // 注册滚动事件监听
}
```

**为什么预留 10px 缓冲？**
- 用户滚动到底部的瞬间可能还未完全等于 `scrollHeight`
- 10px 缓冲确保在用户"接近底部"时就触发，体验更流畅

#### debounce.ts

```typescript
export function debounce(fn: Function, wait: number) {
  let timer: any
  // 闭包变量：定时器 ID，被返回的函数"记住"

  return function (...args: any) {
    // 返回的新函数接收任意参数

    if (timer) clearTimeout(timer)
    // 如果有旧定时器 → 清除（取消上次的延迟执行）

    timer = setTimeout(() => {
      fn(args)  // 等待结束后执行原函数
      clearTimeout(timer)  // ⚠️ 这行是多余的
    }, wait)
  }
}
```

**为什么 `clearTimeout(timer)` 在 setTimeout 内多余？**
- setTimeout 回调执行时，该定时器已自动完成
- 此时 `timer` 变量仍持有旧 ID，但清除一个已触发的定时器是无意义操作
- 虽不影响功能，但属于冗余代码

**为什么 `fn(args)` 应改为 `fn(...args)`？**
- `args` 是数组，`fn(args)` 会将整个数组作为第一个参数传入
- 如果 `fn` 期望多个独立参数，会导致参数错位
- 正确写法：`fn(...args)` 展开数组或 `fn.apply(null, args)`

#### 核心知识点

| 知识点 | 说明 |
|--------|------|
| **闭包** | debounce 内部的 `timer` 被返回函数"记住"，实现跨调用的状态共享 |

| **Composables** | Vue 3 的 `use*` 函数，封装可复用的有状态逻辑，替代 Vue 2 的 mixins |

| **事件清理** | `addEventListener` 必须配对 `removeEventListener`，防止内存泄漏 |

#### 可复用写法

**1. 通用防抖函数（可直接迁移到任何项目）：**
```typescript
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timer: ReturnType<typeof setTimeout> | null = null
  return function (this: any, ...args: Parameters<T>) {
    if (timer !== null) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(this, args)
      timer = null
    }, wait)
  }
}
```

**2. 通用触底加载钩子（可直接迁移到任何项目）：**
```typescript
export default function useLoadMore(
  element: Ref<HTMLElement | null>,
  fn: Function
) {
  function loadMore() {
    const { clientHeight = 0, scrollTop = 0, scrollHeight = 0 } = element.value || {}
    if (clientHeight + scrollTop + 10 >= scrollHeight) fn()
  }
  element.value?.addEventListener('scroll', loadMore)
  onUnmounted(() => element.value?.removeEventListener('scroll', loadMore))
}
```

#### 易错点 / 避坑指南

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| `fn(args)` 参数错位 | `args` 是数组，未展开 | 改用 `fn(...args)` 或 `fn.apply(null, args)` |
| 组件卸载后内存泄漏 | 未移除事件监听 | 在 `onUnmounted` 中调用 `removeEventListener` |
| 防抖函数 `this` 丢失 | 返回的匿名函数改变了 `this` 指向 | 使用 `fn.apply(this, args)` 保持上下文 |
| 类型标注 `function` 小写 | TypeScript 中应为 `Function` 大写 | 使用泛型更精确：`<T extends (...args: any[]) => any>` |
| 滚动判断不触发 | 缓冲值太小或容器未设置高度 | 确保容器有明确高度，缓冲值 ≥ 10px |

#### 7.3.1 import type 与 import 的区别

**问题代码：**
```typescript
// useLoadMore.ts（修复前）
import { Ref } from 'vue'  // ❌ Vite 报错
```

**修复后：**
```typescript
import type { Ref } from 'vue'  // ✅ 类型专用导入
```

**原理拆解：**

```typescript
// import { Ref } 的编译结果（ESM）
import { Ref } from 'vue'
// → 运行时尝试从 vue 模块中查找名为 Ref 的导出
// → Vue 3 的运行产物（vue.runtime.esm-bundler.js）不导出 Ref
// → 报错：does not provide an export named 'Ref'

// import type { Ref } 的编译结果
// → TypeScript 编译阶段直接擦除这行
// → 运行时不存在任何 import 语句
// → 无运行时开销，无报错
```

**判断规则：**

| 导入内容 | 用法 | 示例 |
|----------|------|------|
| 运行时值（函数、变量、组件） | `import { }` | `import { ref, watch } from 'vue'` |
| 纯类型（interface、type、泛型参数） | `import type { }` | `import type { Ref, Note } from 'vue'` |
| 混合（既有值又有类型） | 分开写 | `import { ref } from 'vue'` + `import type { Ref } from 'vue'` |

**TypeScript 配置影响：**
```json
// tsconfig.json
{
  "compilerOptions": {
    "verbatimModuleSyntax": true  // 开启后，纯类型必须用 import type，否则编译报错
  }
}
```

#### 7.3.2 瀑布流分页加载的 items 时序问题

**问题本质：** `items.value = []` 和 `notes.value = res` 是同步执行的，但 Vue 的 DOM 更新是异步的。

**修复前的时序（错误）：**
```
loadMore() 执行
    │
    ├─ items.value = []          ← 同步：清空 DOM 引用
    ├─ notes.value = res         ← 同步：触发 watcher
    │       │
    │       ▼
    │   watcher 触发 initLRlist()
    │       │
    │       ▼
    │   nextTick(() => {
    │       items.value.forEach(...)  ← ⚠️ items 可能为空！
    │   })
    │
    ▼
Vue 开始 DOM 更新（v-for 重新渲染）
    │
    ▼
template ref 回调执行，重新填充 items.value
    │
    ▼
但 nextTick 回调已经执行完了，用的是空的 items
```

**修复后的时序（正确）：**
```
loadMore() 执行
    │
    ├─ notes.value = res         ← 只更新数据，不动 items
    │       │
    │       ▼
    │   watcher 触发 initLRlist()
    │       │
    │       ├─ items.value = []  ← 在 initLRlist 内部清空
    │       │
    │       ▼
    │   nextTick(() => {
    │       items.value.forEach(...)  ← ✅ items 已被 template ref 重新填充
    │   })
    │
    ▼
Vue 开始 DOM 更新（v-for 重新渲染）
    │
    ▼
template ref 回调执行，填充 items.value
    │
    ▼
nextTick 回调执行，使用完整的 items
```

**关键代码对比：**
```typescript
// ❌ 修复前：loadMore 中清空 items
const loadMore = () => {
  stateV.page++
  listStore.getNotesList(stateV.page, stateV.size)
    .then((res) => ((items.value = []), (notes.value = res)))
  //                ↑ 同步清空                ↑ 同步触发 watcher
  // 问题：watcher 的 nextTick 回调执行时，DOM 还没更新完
}

// ✅ 修复后：initLRlist 中清空 items
const loadMore = () => {
  stateV.page++
  listStore.getNotesList(stateV.page, stateV.size)
    .then((res) => (notes.value = res))  // 只更新数据
}

const initLRlist = () => {
  // ...
  items.value = []       // 清空后，Vue 会触发 .init-list 的 v-for 重新渲染
  nextTick(() => {       // 等待 DOM 更新完成
    items.value.forEach(...)  // 此时 template ref 已重新收集
  })
}
```

**为什么 `items.value = []` 放在 `initLRlist` 内部就正确？**

因为 `initLRlist` 是被 `watch(notes, ...)` 触发的。当 watcher 执行时：
1. `items.value = []` 同步清空 → Vue 标记 `.init-list` 需要重新渲染
2. `nextTick` 注册回调 → 等待本轮 DOM 更新完成
3. Vue 执行 DOM 更新 → `v-for` 重新渲染所有 `notes` 项 → template ref 回调执行 → `items.value` 被重新填充
4. `nextTick` 回调执行 → 拿到完整的 `items.value`

#### 7.3.3 Pinia store 的分页数据累积模式

**store 代码：**
```typescript
// notelist.ts
const list = ref<Note[]>([])

const getNotesList = async (page: number, size: number) => {
  const res = await getNotes<NoteList>(page, size)
  if (page === 1) {
    list.value = res         // 第一页：直接替换
  } else {
    list.value.push(...res)  // 后续页：追加到已有数据
  }
  return [...list.value]     // 返回完整累积数据
}
```

**数据流：**
```
第 1 次调用 getNotesList(1, 13)
    │
    ├─ API 返回 13 条
    ├─ page === 1 → list.value = [1~13]
    └─ return [1~13]          → 组件 notes.value = [1~13]

第 2 次调用 getNotesList(2, 13)
    │
    ├─ API 返回 13 条
    ├─ page !== 1 → list.value.push(...[14~26])
    └─ return [1~26]          → 组件 notes.value = [1~26]

第 3 次调用 getNotesList(3, 13)
    │
    ├─ API 返回 13 条
    ├─ page !== 1 → list.value.push(...[27~39])
    └─ return [1~39]          → 组件 notes.value = [1~39]
```

**设计权衡：**

| 方案 | 优点 | 缺点 |
|------|------|------|
| store 内累积（当前方案） | 组件逻辑简单，store 统一管理数据 | store 的 `list` 持续增长，内存占用 |
| 组件内累积 | store 无状态，职责单一 | 组件需要维护 `notes.value = [...notes.value, ...res]` |
| 仅返回当前页 | 内存最优 | 组件需要自行合并，瀑布流重排成本高 |

**为什么选择 store 内累积？**
- `initNote()` 编辑回显需要从 `listStore.list` 中查找笔记内容
- 如果 store 不累积，切换到编辑页时 store 只有最后一页数据，找不到目标笔记

#### 7.3.4 notes（数据层）与 items（DOM 引用层）的职责分工

**两个 ref 的定义：**
```typescript
const notes = ref([] as NoteList)      // 数据层：存储笔记内容
const items = ref([] as HTMLElement[])  // DOM 层：存储元素引用
```

**职责对比：**

| 维度 | `notes` | `items` |
|------|---------|---------|
| 内容 | `{ _id, content, dates }` 数据对象 | `HTMLElement` DOM 元素 |
| 来源 | API 请求 → store → 组件 | template ref 回调自动收集 |
| 用途 | 渲染列表内容、填充左右列 | 测量 `clientHeight`，计算瀑布流分配 |
| 变化时机 | API 返回新数据时 | `notes` 变化 → DOM 更新 → ref 回调触发 |
| 清空时机 | 不主动清空（由 store 管理） | `initLRlist` 中 `nextTick` 前清空 |

**协作关系图：**
```
API 数据 ──→ notes.value ──→ v-for 渲染 .init-list ──→ template ref ──→ items.value
                                                                       │
                                                                       ▼
                                                              items[i].clientHeight
                                                                       │
                                                                       ▼
                                                              瀑布流分配算法
                                                                       │
                                                              ┌────────┴────────┐
                                                              ▼                  ▼
                                                        state.leftList      state.rightList
                                                              │                  │
                                                              ▼                  ▼
                                                        左列 v-for           右列 v-for
```

**为什么需要两个独立的 ref？**
- `notes` 存数据，`items` 存 DOM → 数据和视图分离
- 瀑布流分配需要 `clientHeight`（DOM 属性），只有拿到真实 DOM 元素才能测量
- 如果只用 `notes`，无法得知每条笔记渲染后的实际高度
- 如果只用 `items`，没有数据内容来填充左右列

```
    │       ├─ API 返回 13 条新数据
    │       ├─ store: list.value.push(...res)  （追加到 store）
    │       └─ return [...list.value]          （返回全部累积数据）
    │
    ├─ notes.value = res                （组件拿到 26 条数据）
    │
    ▼
watch(notes) 触发 → initLRlist()
    │
    ├─ items.value = []                 （清空 DOM 引用）
    │
    ▼
nextTick() 等待 DOM 更新
    │
    ├─ .init-list 的 v-for 重新渲染 26 个元素
    ├─ template ref 回调执行 → items.value 收集 26 个 HTMLElement
    │
    ▼
nextTick 回调执行
    │
    ├─ items.value.forEach() 遍历 26 个元素
    ├─ 测量每个元素的 clientHeight
    ├─ 贪心分配到左列/右列

    │
    ▼
模板重新渲染左右列 → 瀑布流高度增长 → 可继续滚动
```

### 7.5 总结笔记（增量）

#### 新增关键认知

| # | 认识 | 来源 |
|---|------|------|
| 1 | `import type` 在编译后完全擦除，纯类型导入必须用它，否则 Vite/Webpack 会尝试运行时解析 | `Ref` 报错修复 |
| 2 | Vue 的 DOM 更新是异步的，清空响应式数组和设置新数据不能在同一同步块内完成，否则 `nextTick` 拿到的是空数组 | `items` 时序修复 |
| 3 | Pinia store 可以内部维护累积状态，组件只需 `notes.value = res`，不需要自行合并分页数据 | `getNotesList` 设计 |
| 4 | `notes`（数据）和 `items`（DOM）分离是瀑布流的核心——数据填充内容，DOM 测量高度 | 双 ref 架构 |
| 5 | Vite HMR 有时缓存旧模块，新增 store 方法后不生效 → 重启 dev server | `updateNoteList` 报错 |

#### 可复用写法

**1. 类型安全的 Composable 参数签名：**
```typescript
import type { Ref } from 'vue'

export default function useLoadMore(
  element: Ref<HTMLElement | null>,
  fn: () => void
) {
  function loadMore() {
    const { clientHeight = 0, scrollTop = 0, scrollHeight = 0 } = element.value || {}
    if (clientHeight + scrollTop >= scrollHeight - 5) fn()
  }
  element.value?.addEventListener('scroll', loadMore)
}
```

**2. 安全的瀑布流重排模式（清空 + nextTick + 重分配）：**
```typescript
const initLRlist = () => {
  const leftArr: Note[] = []
  const rightArr: Note[] = []
  let leftH = 0, rightH = 0

  items.value = []          // 清空 → 触发 DOM 重新渲染
  nextTick(() => {          // 等待 DOM 更新 + template ref 收集完成
    items.value.forEach((el, i) => {
      if (leftH <= rightH) {
        leftArr.push(notes.value[i])
        leftH += el.clientHeight
      } else {
        rightArr.push(notes.value[i])
        rightH += el.clientHeight
      }
    })
    state.leftList = leftArr.reverse()
    state.rightList = rightArr.reverse()
  })
}
```

#### 新增易错点与避坑建议

| # | 问题 | 原因 | 解决方案 |
|---|------|------|----------|
| 1 | Vite 报 `does not provide an export named 'Ref'` | `Ref` 是类型，不是运行时值 | 改用 `import type { Ref }` |
| 2 | 分页加载后瀑布流不增长、无法继续滚动 | `items.value = []` 和 `notes.value = res` 同步执行，nextTick 时 items 为空 | 将 `items.value = []` 移到 `initLRlist` 内部的 `nextTick` 之前 |
| 3 | store 新增方法后组件调用报 `is not a function` | Vite HMR 缓存旧模块 | 重启 dev server |
| 4 | 编辑页回显找不到笔记数据 | store 未累积，只有最后一页 | store 内部 `push` 追加，返回完整 `list` |
| 5 | `debounce` 中 `fn(args)` 参数错位 | `args` 是数组，未展开 | 改用 `fn(...args)` |

#### 修订点清单

| 修订点 | 原结论 | 新结论 | 触发原因 |
|--------|--------|--------|----------|
| 1 | `useLoadMore.ts` 使用 `import { Ref }` | 改为 `import type { Ref }` | Vite ESM 解析不支持运行时导入纯类型 |
| 2 | `loadMore` 中 `items.value = []` 同步清空 | 移到 `initLRlist` 内部，`nextTick` 前清空 | 异步 DOM 更新导致 nextTick 时 items 为空 |
| 3 | `handleSearch` / `handleClear` 中 `items.value = []` | 同样去掉，由 `initLRlist` 统一管理 | 与 `loadMore` 同一时序问题 |
